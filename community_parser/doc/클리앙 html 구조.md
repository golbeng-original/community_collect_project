                    <span class="shortname fixed">카테고리 숏 이름</span>

                    <!-- Aswome Icon 사용-->

                        <span class="rSymph05">댓글 수</span>
                    </a>
                </a>
            </div>
            <div class="list_author">
                <span class="nickname">
                    <span>
                        <!-- img 닉네임-->
